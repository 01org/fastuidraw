30 /*!\def FASTUIDRAW_CHARACTER_ENCODING_VALUE
31  * Conveniance macro to define a 32-bit character encoding scheme
32  * from four unsigned 8-bit values.
33  */
34 #define FASTUIDRAW_CHARACTER_ENCODING_VALUE(a, b, c, d) \
35  (uint32_t(a) << 24u) | (uint32_t(b) << 16u) | (uint32_t(c) << 8u) | uint32_t(d)
