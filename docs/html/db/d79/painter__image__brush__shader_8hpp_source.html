263  /*!

265  * Enumeration describing the roles of the bits for
266  * the sub-shader ID's.
267  */
268  enum sub_shader_bits
269  {
270  /*!
271  * Number of bits needed to encode filter for image.
272  * A value of 0 indicates no image applied, a non-zero
273  * value indicates an image applied and the value
274  * specifies what filter via the enumeration \ref
275  * filter_t.
276  */
277  filter_num_bits = 2,
278 
279  /*!
280  * Number of bits used to encode number of mipmap
281  * levels (when an image is present).
282  */
283  mipmap_num_bits = 7,
284 
285  /*!
286  * Number of bits needed to encode the image type
287  * (when an image is present). The possible values
288  * are given by the enumeration \ref Image::type_t.
289  */
290  type_num_bits = 4,
291 
292  /*!
293  * Number of bits needed to encode the value of
294  * Image::format().
295  */
296  format_num_bits = 1,
297 
298  /*!
299  * first bit for if image is present on the brush and if so, what filter
300  */
301  filter_bit0 = 0,
302 
303  /*!
304  * first bit to indicate maximum mipmap level to use
305  */
306  mipmap_bit0 = filter_bit0 + filter_num_bits,
307 
308  /*!
309  * First bit to hold the type of image present if an image is present;
310  * the value is the enumeration in \ref Image::type_t
311  */
312  type_bit0 = mipmap_bit0 + mipmap_num_bits,
313 
314  /*!
315  * First bit to encode \ref Image::format_t
316  */
317  format_bit0 = type_bit0 + type_num_bits,
318 
319  /*!
320  * The total number of bits needed to specify the sub-shader IDs.
321  */
322  number_bits = format_bit0 + format_num_bits,
323 
324  /*!
325  * the total number of sub-shaders
326  */
327  number_sub_shaders = 1u << number_bits
328  };
