29 /*!\def FASTUIDRAWnew
30  * When creating FastUIDraw objects, one must use FASTUIDRAWnew instead of new
31  * to create objects. For debug build of FastUIDraw, allocations with FASTUIDRAWnew
32  * are tracked and at program exit a list of those objects not deleted by
33  * FASTUIDRAWdelete are printed with the file and line number of the allocation.
34  * For release builds of FastUIDraw, allocations are not tracked and std::malloc
35  * is used to allocate memory. Do NOT use FASTUIDRAWnew for creating arrays
36  * (i.e. p = new type[N]) as FASTUIDRAWdelete does not handle array deletion.
37  */
38 #define FASTUIDRAWnew \
39  ::new(__FILE__, __LINE__)
40 
41 /*!\def FASTUIDRAWdelete
42  * Use \ref FASTUIDRAWdelete to delete objects that were allocated with \ref
43  * FASTUIDRAWnew. For debug builds of FastUIDraw, if the memory was not tracked
44  * an error message is emitted.
45  * \param ptr address of object to delete, value must be a return value
46  * from FASTUIDRAWnew
47  */
48 #define FASTUIDRAWdelete(ptr) \
49  do { \
50  fastuidraw::memory::check_object_exists(ptr, __FILE__, __LINE__); \
51  fastuidraw::memory::call_dtor(ptr); \
52  fastuidraw::memory::free_implement(ptr, __FILE__, __LINE__); \
53  } while(0)
