227  /*!
228  * Compute pixel coordinates from normalized device coords
229  * using this Viewport values. The pixel coordinates are so
230  * that (0, 0) is the bottom left.
231  * \param ndc normalized device coordinates
232  */
233  vec2
234  compute_pixel_coordinates(vec2 ndc) const
235  {
236  ndc += vec2(1.0f); // place in range [0, 2]
237  ndc *= 0.5f; // normalize to [0, 1]
238  ndc *= vec2(m_dimensions); // normalize to dimension
239  ndc += vec2(m_origin); // translate to origin
240  return ndc;
241  }
242 
243  /*!
244  * Compute normalized device coordinates from pixel
245  * coordinates.
246  * \param pixel pixel coordinates where (0, 0) corresponds to bottom
247  * left of the surface
248  */
249  vec2
250  compute_normalized_device_coords(vec2 pixel) const
251  {
252  pixel -= vec2(m_origin); // translate from origin
253  pixel /= vec2(m_dimensions); // normalize to [0, 1]
254  pixel *= 2.0f; // normalize to [0, 2]
255  pixel -= vec2(1.0f); // palce in range [-1, 1]
256  return pixel;
257  }
