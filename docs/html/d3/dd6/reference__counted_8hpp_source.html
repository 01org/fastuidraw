38  /*!
39  * \brief
40  * A wrapper over a pointer to implement reference counting.
41  *
42  * The class T must implement the static methods
43  * - T::add_reference(const T*)
44  * - T::remove_reference(const T*)
45  *
46  * where T::add_reference() increment the reference count and
47  * T::remove_reference() decrements the reference count and will

413  /*!
414  * \brief
415  * Base class to use for reference counted objects,
416  * for using reference_counted_ptr. See also \ref reference_counted.
417  * Object deletion (when the reference count goes to zero) is performed
418  * via \ref FASTUIDRAWdelete. As a consequence of using \ref
419  * FASTUIDRAWdelete, objects must be created with \ref FASTUIDRAWnew.
420  *
421  * \tparam T object type that is reference counted
422  * \tparam Counter object type to perform reference counting.
423  *
424  * The type Counter must expose the methods:
425  * - void add_reference() to increment the counter
426  * - bool remove_reference() to decrement the counter and return true
427  * if the counter is zero after the decrement operation.
428  */
429  template<typename T, typename Counter>
430  class reference_counted_base:noncopyable
431  {
432  public:
433  /*!
434  * Empty ctor.
435  */
436  reference_counted_base(void)
437  {}
438 
439  virtual
440  ~reference_counted_base()
441  {}

443  /*!
444  * Adds a reference count to an object.
445  * \param p pointer to object to which to add reference count.
446  */
447  static
448  void
449  add_reference(const reference_counted_base<T, Counter> *p)
450  {
451  FASTUIDRAWassert(p);
452  p->m_counter.add_reference();
453  }
454 
455  /*!
456  * Removes a reference count to an object, if the reference
457  * count is 0, then deletes the object with \ref FASTUIDRAWdelete.
458  * \param p pointer to object from which to remove reference count.
459  */
460  static
461  void
462  remove_reference(const reference_counted_base<T, Counter> *p)
463  {
464  FASTUIDRAWassert(p);
465  if (p->m_counter.remove_reference())
466  {
467  reference_counted_base<T, Counter> *q;
468  q = const_cast<reference_counted_base<T, Counter>*>(p);
469  FASTUIDRAWdelete(q);
470  }
471  }
472  private:
473  mutable Counter m_counter;
474  };
