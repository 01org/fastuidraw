75  on_atlas,

77  /*!
78  * Indicates that the Image is backed by a gfx API
79  * texture via a bindless interface. Such images
80  * do not introduce draw-breaks and are simple
81  * from. The only draw back, is that not all 3D
82  * API implementations support bindless texruing
83  * (for example the GL/GLES backends require that
84  * the GL/GLES implementation support an extension.
85  */
86  bindless_texture2d,
87 
88  /*!
89  * Indicates to source the Image data from a currently
90  * bound texture of the 3D API context. These image are
91  * simple to sample from but introduce draw breaks. Using
92  * images of these types should be avoided at all costs
93  * since whenever one uses Image objects of these types
94  * the 3D API state needs to change which induces a stage
95  * change and draw break, harming performance.
96  */
97  context_texture2d,
98  };
