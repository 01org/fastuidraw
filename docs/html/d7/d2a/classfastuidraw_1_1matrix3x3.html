void fastuidraw::matrix3x3< T >::rotate ( T  angle)
inline

Apply a rotation matrix on the right of the matrix, algebraicly equivalent to

matrix3x3 M;
s = t_sin(angle);
c = t_cos(angle);
M(0, 0) = c;
M(1, 0) = s;
M(0, 1) = -s;
M(1, 1) = c;
*this = *this * M;
