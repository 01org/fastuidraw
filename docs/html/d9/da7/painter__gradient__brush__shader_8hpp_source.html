33  /*!
34  * \brief
35  * A \ref PainterGradientBrushShader represents a set of brush shaders to
36  * to perform a gradient. Internally it containts four generic parent shaders:
37  * - a parent shader able to handle any gradient and spread type
38  * - a parent shader only for linear gradients able to handle any spread type
39  * - a parent shader only for radial gradients able to handle any spread type
40  * - a parent shader only for sweep gradients able to handle any spread type
41  */

47  /*!
48  * \brief
49  * Enumeration describing the roles of the bits for
50  * the sub-shader ID's.

52  enum sub_shader_bits
53  {
54  /*!
55  * Number of bits used to encode a \ref spread_type_t
56  */
57  spread_type_num_bits = 2,
58 
59  /*!
60  * Number of bits used to encode the gradient type,
61  * see \ref PainterBrushEnums::gradient_type_t
62  */
63  gradient_type_num_bits = 2,
64 
65  /*!
66  * first bit used to encode the \ref spread_type_t
67  */
68  spread_type_bit0 = 0,
69 
70  /*!
71  * first bit used to encode the \ref PainterBrushEnums::gradient_type_t
72  */
73  gradient_type_bit0 = spread_type_bit0 + spread_type_num_bits,
74 
75  /*!
76  * The total number of bits needed to specify the sub-shader IDs.

78  number_bits = gradient_type_bit0 + gradient_type_num_bits,

86  /*!

90  number_sub_shaders_of_specific_gradient = 1u << spread_type_num_bits
91  };
