294  /*!
295  * Sets the brush to have a sweep gradient where the sign is
296  * determined by a PainterEnums::screen_orientation and a
297  * PainterEnums::rotation_orientation_t.
298  * \param cs color stops for gradient. If handle is invalid,
299  * then sets brush to not have a gradient.
300  * \param p position of gradient
301  * \param theta angle of the sweep gradient, this value
302  * should be in the range [-PI, PI]
303  * \param F the repeat factor applied to the interpolate,
304  * a negative reverses the orientation of the sweep.
305  * \param orientation orientation of the screen
306  * \param rotation_orientation orientation of the sweep
307  */
308  PainterGradientBrushShaderData&
309  sweep_gradient(const reference_counted_ptr<const ColorStopSequenceOnAtlas> &cs,
310  const vec2 &p, float theta,
311  enum PainterEnums::screen_orientation orientation,
312  enum PainterEnums::rotation_orientation_t rotation_orientation,
313  float F)
314  {
315  float S;
316  bool b1(orientation == PainterEnums::y_increases_upwards);
317  bool b2(rotation_orientation == PainterEnums::counter_clockwise);
318 
319  S = (b1 == b2) ? 1.0f : -1.0f;
320  return sweep_gradient(cs, p, theta, S * F);
321  }

387  /*!
388  * \brief
389  * A \ref PainterGradientBrushShader represents a set of brush shaders to
390  * to perform a gradient. Internally it containts four generic parent shaders:
391  * - a parent shader able to handle any gradient and spread type
392  * - a parent shader only for linear gradients able to handle any spread type
393  * - a parent shader only for radial gradients able to handle any spread type
394  * - a parent shader only for sweep gradients able to handle any spread type
395  */
396  class PainterGradientBrushShader:
397  public reference_counted<PainterGradientBrushShader>::concurrent,
398  public PainterBrushEnums
399  {
400  public:
401  /*!
402  * \brief
403  * Enumeration describing the roles of the bits for
404  * the sub-shader ID's.
405  */
406  enum sub_shader_bits
407  {
408  /*!
409  * Number of bits used to encode a \ref spread_type_t
410  */
411  spread_type_num_bits = 2,
412 
413  /*!
414  * Number of bits used to encode the gradient type,
415  * see \ref PainterBrushEnums::gradient_type_t
416  */
417  gradient_type_num_bits = 2,
418 
419  /*!
420  * first bit used to encode the \ref spread_type_t
421  */
422  spread_type_bit0 = 0,
423 
424  /*!
425  * first bit used to encode the \ref PainterBrushEnums::gradient_type_t
426  */
427  gradient_type_bit0 = spread_type_bit0 + spread_type_num_bits,
428 
429  /*!
430  * The total number of bits needed to specify the sub-shader IDs.
431  */
432  number_bits = gradient_type_bit0 + gradient_type_num_bits,
433 
434  /*!
435  * the total number of sub-shaders that the generic parent
436  * shader has.
437  */
438  number_sub_shaders_of_generic_gradient = 1u << number_bits,
439 
440  /*!
441  * the total number of sub-shaders that a parent shader for
442  * a specific gradient type has.
443  */
444  number_sub_shaders_of_specific_gradient = 1u << spread_type_num_bits
445  };
446 
447  /*!
448  * Various bit-mask values derived from \ref sub_shader_bits
449  */
450  enum sub_shader_masks
451  {
452  /*!
453  * mask generated from \ref spread_type_bit0 and \ref spread_type_num_bits
454  */
455  gradient_spread_type_mask = FASTUIDRAW_MASK(spread_type_bit0, spread_type_num_bits),

457  /*!
458  * mask generated from \ref gradient_type_bit0 and \ref gradient_type_num_bits
