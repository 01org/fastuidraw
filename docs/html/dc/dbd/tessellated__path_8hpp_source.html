43 /*!
44  * \brief
45  * An TessellatedPath represents the tessellation of a Path
46  * into line segments and arcs.
47  *
48  * A single contour of a TessellatedPath is constructed from
49  * a single \ref PathContour of the source \ref Path. Each
50  * edge of a contour of a TessellatedPath is contructed from
51  * a single \ref PathContour::interpolator_base of the source \ref
52  * PathContour. The ordering of the contours of a
53  * TessellatedPath is the same ordering as the source
54  * \ref PathContour objects of the source \ref Path. Also,
55  * the ordering of edges within a contour is the same ordering
56  * as the \ref PathContour::interpolator_base objects of
57  * the source \ref PathContour. In particular, for each contour
58  * of a TessellatedPath, if an edge is closed, the closing edge
59  * is the last edge.
60  */
61 class TessellatedPath:
62  public reference_counted<TessellatedPath>::non_concurrent
63 {
64 public:

70  {
71  /*!
72  * Indicates that the segment is an arc segment,
73  * i.e. it connects two point via an arc of a
74  * circle
75  */
76  arc_segment,
77 
78  /*!
79  * Indicates that the segment is a line segment
80  * i.e. it connects two point via a line.
81  */
82  line_segment,
83  };
