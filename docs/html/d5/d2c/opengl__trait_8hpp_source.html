46 /*!
47  * \brief
48  * Type trait struct that provides type information to feed GL commands.
49  * \tparam T type from which to extract values. The struct is specicialized for
50  * each of the GL types: GLubyte, GLbyte, GLuint, GLint, GLushort, GLshort, GLfloat
51  * and recursively for vecN
52  */
53 template<typename T>
54 struct opengl_trait
55 {
56  /*!
57  * \brief
58  * Typedef to template parameter
59  */
60  typedef T data_type;
61 
62  /*!
63  * \brief
64  * For an array type, such as vecN,
65  * element type of the array, otherwise
66  * is same as data_type. Note,
67  * for vecN types of vecN's reports
68  * the element type of inner array type,
69  * for example
70  * \code vecN<vecN<something,N>, M> \endcode
71  * gives something for basic_type
72  */
73  typedef T basic_type;
74 
75  enum
76  {
77  /*!
78  * GL type label, for example if basic_type
79  * is GLuint, then type is GL_UNSIGNED_INT
80  */
81  type = GL_INVALID_ENUM
82  };
83 
84  enum
85  {
86  /*!
87  * The number of basic_type
88  * elements packed into one data_type
89  */
90  count = 1
91  };
92 
93  enum
94  {
95  /*!
96  * The space between adjacent
97  * data_type elements in an array
98  */
99  stride = sizeof(T)
100  };
101 };
