42  /*!
43  * \brief
44  * A PainterBrush defines a brush for painting via Painter.
45  *
46  * The brush applies, in the following order:
47  * -# a constant color, specified by \ref color()
48  * -# optionally modulates by an image, specified by \ref
49  * image() or \ref sub_image() and \ref no_image()
50  * -# optionally modulates by a gradient, see \ref
51  * linear_gradient(), \ref radial_gradient(),
52  * sweep_gradient() and \ref no_gradient().
53  *
54  * An item shader's vertex stage provides the coordinate
55  * fed to the brush. That coordinate is procesed in the
56  * following order before it is fed to the image and gradient:
57  * -# an optional 2x2 matrix is applied, see \ref
58  * transformation_matrix(), \ref apply_matrix()
59  * \ref apply_shear(), \ref apply_rotate()
60  * and \ref no_transformation_matrix()
61  * -# an optional translation is applied specified, see
62  * \ref transformation_translate(), apply_matrix()
63  * \ref apply_translate() and \ref no_transformation_translation()
64  * -# an optional repeat window is applied, see \ref
65  * repeat_window() and \ref no_repeat_window().
66  */

69  public PainterBrushEnums

77  enum feature_bits

79  /*!
80  * Number of bits needed to encode if and how
81  * an \ref Image is sourced from.
82  */
83  image_num_bits = PainterImageBrushShader::number_bits,
84

87  * see \ref PainterBrushEnums::gradient_type_t

90 
91  /*!
92  * First bit to encode if and how the brush sources
93  * from an \ref Image
94  */
95  image_bit0 = 0,

97  /*!
98  * first bit used to encode the \ref PainterBrushEnums::gradient_type_t
99  */
100  gradient_bit0 = image_bit0 + image_num_bits,
101 
102  /*!
103  * Bit up if the brush has a repeat window
104  */
105  repeat_window_bit = gradient_bit0 + gradient_num_bits,
106 
107  /* First bit used to encore the spread mode for the x-coordinate
108  * if a repeat window is active
109  */
110  repeat_window_x_spread_type_bit0,
111 
112  /* First bit used to encore the spread mode for the x-coordinate
113  * if a repeat window is active
114  */
115  repeat_window_y_spread_type_bit0 = repeat_window_x_spread_type_bit0 + PainterGradientBrushShader::spread_type_num_bits,
116 
117  /*!
118  * Bit up if transformation 2x2 matrix is present
119  */
120  transformation_translation_bit = repeat_window_y_spread_type_bit0 + PainterGradientBrushShader::spread_type_num_bits,
121 
122  /*!
123  * Bit up is translation is present
124  */
125  transformation_matrix_bit,
126 
127  /*!
128  * Must be last enum, gives number of bits needed to hold feature bits
129  * of a PainterBrush.
130  */
131  number_feature_bits = transformation_matrix_bit + 1,
