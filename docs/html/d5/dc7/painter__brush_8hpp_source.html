38  /*!
39  * \brief
40  * A PainterBrush defines a brush for painting via Painter.
41  *
42  * The brush applies, in the following order:
43  * -# a constant color, specified by \ref color()
44  * -# optionally modulates by an image, specified by \ref
45  * image() or \ref sub_image() and \ref no_image()
46  * -# optionally modulates by a gradient, see \ref
47  * linear_gradient(), \ref radial_gradient(),
48  * sweep_gradient() and \ref no_gradient().
49  *
50  * An item shader's vertex stage provides the coordinate
51  * fed to the brush. That coordinate is procesed in the
52  * following order before it is fed to the image and gradient:
53  * -# an optional 2x2 matrix is applied, see \ref
54  * transformation_matrix(), \ref apply_matrix()
55  * \ref apply_shear(), \ref apply_rotate()
56  * and \ref no_transformation_matrix()
57  * -# an optional translation is applied specified, see
58  * \ref transformation_translate(), apply_matrix()
59  * \ref apply_translate() and \ref no_transformation_translation()
60  * -# an optional repeat window is applied, see \ref
61  * repeat_window() and \ref no_repeat_window().
62  */

91  * Enumeration to specify what kind of gradient is applied

93  enum gradient_type_t
94  {
95  /*!
96  * Indicates that no gradient is present
97  */
98  no_gradient_type = 0,
99 
100  /*!
101  * Indicates that a linear gradient is present;
102  * a linear gradient is defined by two points
103  * p0 and p1 where the interpolate at a point p
104  * is the value t when p = p0 + (p1 - p0) * t
105  */
106  linear_gradient_type = 1,
107 
108  /*!
109  * Indicates that a radial gradient is present;
110  * a radial gradient is defined by two circles
111  * C0 = Circle(p0, r0), C1 = Circle(p1, r1)
112  * where the interpolate at a point p is the
113  * time t when p is on the circle C(t) where
114  * C(t) = Circle(p(t), r(t)) where
115  * p(t) = p0 + (p1 - p0) * t
116  * r(t) = r0 + (r1 - r0) * t
117  */
118  radial_gradient_type = 2,
119 
120  /*!
121  * A sweep gradient is defined by a single point C,
122  * an angle theta (in radians), a sign S and a factor F.
123  * The angle theta represents at what angle the gradient
124  * starts, the point C is the center point of the
125  * sweep, the sign of S represents the angle orientation
126  * and the factor F reprsents how many times the
127  * gradient is to be repated. Precisely, the interpolate
128  * at a point p is defined as t_interpolate where
129  * \code
130  * vec2 d = p - C;
131  * float theta, v;
132  * theta = S * atan(p.y, p.x);
133  * if (theta < alpha )
134  * {
135  * theta += 2 * PI;
136  * }
137  * theta -= alpha;
138  * v = (theta - angle) / (2 * PI);
139  * t_interpolate = (S < 0.0) ? F * (1.0 - v) : F * v;
140  * \endcode
141  */
142  sweep_gradient_type = 3,
143  };
