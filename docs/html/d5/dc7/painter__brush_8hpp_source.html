38  /*!
39  * \brief
40  * A PainterBrush defines a brush for painting via Painter.
41  *
42  * The brush applies, in the following order:
43  * -# a constant color, specified by \ref pen()
44  * -# optionally applies an image, specified by \ref image() or
45  * \ref sub_image() also see \ref no_image()
46  * -# optionally applies a linear or radial gradient, see \ref
47  * linear_gradient() and \ref radial_gradient() also see
48  * \ref no_gradient().
49  *
50  * An item shader's vertex stage provides the coordinate
51  * fed to the brush. That coordinate is procesed in the
52  * following order before it is fed to the image and gradient:
53  * -# an optional 2x2 matrix is applied specified by \ref
54  * transformation_matrix(), also see \ref
55  * no_transformation_matrix() and \ref
56  * transformation().
57  * -# an optional translation is applied specified by \ref
58  * transformation_translate(), also see \ref
59  * no_transformation_translation() and \ref
60  * transformation().
61  * -# an optional repeat window is applied specified by \ref
62  * repeat_window(), also see no_repeat_window().
63  */

96  /*!
97  * \brief
98  * Enumeration to specify what kind of gradient is applied
99  */
100  enum gradient_type_t
101  {
102  /*!
103  * Indicates that no gradient is present
104  */
105  no_gradient_type = 0,
106 
107  /*!
108  * Indicates that a linear gradient is present;
109  * a linear gradient is defined by two points
110  * p0 and p1 where the interpolate at a point p
111  * is the value t when p = p0 + (p1 - p0) * t
112  */
113  linear_gradient_type = 1,
114 
115  /*!
116  * Indicates that a radial gradient is present;
117  * a radial gradient is defined by two circles
118  * C0 = Circle(p0, r0), C1 = Circle(p1, r1)
119  * where the interpolate at a point p is the
120  * time t when p is on the circle C(t) where
121  * C(t) = Circle(p(t), r(t)) where
122  * p(t) = p0 + (p1 - p0) * t
123  * r(t) = r0 + (r1 - r0) * t
124  */
125  radial_gradient_type = 2,
126 
127  /*!
128  * A sweep gradient is defined by a single point C,
129  * an angle theta (in radians), a sign S and a factor F.
130  * The angle theta represents at what angle the gradient
131  * starts, the point C is the center point of the
132  * sweep, the sign of S represents the angle orientation
133  * and the factor F reprsents how many times the
134  * gradient is to be repated. Precisely, the interpolate
135  * at a point p is defined as t_interpolate where
136  * \code
137  * vec2 d = p - C;
138  * float theta, v;
139  * theta = S * atan(p.y, p.x);
140  * if (theta < alpha )
141  * {
142  * theta += 2 * PI;
143  * }
144  * theta -= alpha;
145  * v = (theta - angle) / (2 * PI);
146  * t_interpolate = (S < 0.0) ? F * (1.0 - v) : F * v;
147  * \endcode
148  */
149  sweep_gradient_type = 3,
150  };
