280  {

282  blend_w3c_multiply, /*!< W3C multiply mode: Dest * Src */
283  blend_w3c_screen, /*!< W3C screen mode: 1 - (1 - Dest) * (1 - Src) */
284 
285  /*!
286  * W3C overlay mode: for each channel,
287  * (Dst <= 0.5) ?
288  * 2.0 * Dest * Src :
289  * 1 - 2 * (1 - Dst) * (1 - Src)
290  */
291  blend_w3c_overlay,
292  blend_w3c_darken, /*!< W3C darken mode: min(Src, Dest) */
293  blend_w3c_lighten, /*!< W3C lighten mode: max(Src, Dest) */
294 
295  /*!
296  * W3C color-dodge mode: for each channel
297  * (Dest == 0) ? 0 : (Src == 1) ? 1 : min(1, Dst / (1 - Src) )
298  * i.e. if Dest is 0, write 0. If Src is 1, write 1. Otherwise
299  * write Dst / (1 - Src)
300  */
301  blend_w3c_color_dodge,
302 
303  /*!
304  * W3C color-burn mode: for each channel
305  * (Dest == 1) ? 1 : (Src == 0) ? 0 : 1 - min(1, (1 - Dst) / Src)
306  * i.e. if Dest is 1, write 1. If Src is 0, write 0. Otherwise
307  * write (1 - Dst ) / Src
308  */
309  blend_w3c_color_burn,
310 
311  /*!
312  * W3C hardlight mode: for each channel,
313  * (Src <= 0.5) ?
314  * 2.0 * Dest * Src :
315  * 1 - 2 * (1 - Dst) * (1 - Src)
316  */
317  blend_w3c_hardlight,
318 
319  /*!
320  * W3C soft light mode: for each channel:
321  * (Src <= 0.5) ?
322  * Dst - (1 - 2 * Src) * Dst * (1 - Dst) :
323  * Dst + (2 * Src - 1) * (Z - Dst)
324  * where
325  * Z = (Dst <= 0.25) ?
326  * ((16 * Dst - 12) * Dst + 4) * Dst :
327  * sqrt(Dst)
328  */
329  blend_w3c_softlight,
330 
331  blend_w3c_difference, /*!< W3C difference mode: for each channel, abs(Dest - Src) */
332  blend_w3c_exclusion, /*!< W3C exclusion mode: for each channel, Dest + Src - 2 * Dest * Src */
333
