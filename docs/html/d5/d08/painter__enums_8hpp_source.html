193  shader_anti_alias_high_quality,

198  */
199

283  blend_w3c_lighten, /*!< W3C lighten mode: max(Src, Dest) */
284 
285  /*!
286  * W3C color-dodge mode: for each channel
287  * (Dest == 0) ? 0 : (Src == 1) ? 1 : min(1, Dst / (1 - Src) )
288  * i.e. if Dest is 0, write 0. If Src is 1, write 1. Otherwise
289  * write Dst / (1 - Src)
290  */
291  blend_w3c_color_dodge,
292 
293  /*!
294  * W3C color-burn mode: for each channel
295  * (Dest == 1) ? 1 : (Src == 0) ? 0 : 1 - min(1, (1 - Dst) / Src)
296  * i.e. if Dest is 1, write 1. If Src is 0, write 0. Otherwise
297  * write (1 - Dst ) / Src
298  */
299  blend_w3c_color_burn,
300 
301  /*!
302  * W3C hardlight mode: for each channel,
303  * (Src <= 0.5) ?
304  * 2.0 * Dest * Src :
305  * 1 - 2 * (1 - Dst) * (1 - Src)
306  */
307  blend_w3c_hardlight,
308 
309  /*!
310  * W3C soft light mode: for each channel:
311  * (Src <= 0.5) ?
312  * Dst - (1 - 2 * Src) * Dst * (1 - Dst) :
313  * Dst + (2 * Src - 1) * (Z - Dst)
314  * where
315  * Z = (Dst <= 0.25) ?
316  * ((16 * Dst - 12) * Dst + 4) * Dst :
317  * sqrt(Dst)
318  */
319  blend_w3c_softlight,
320 
321  blend_w3c_difference, /*!< W3C difference mode: for each channel, abs(Dest - Src) */
322  blend_w3c_exclusion, /*!< W3C exclusion mode: for each channel, Dest + Src - 2 * Dest * Src */
