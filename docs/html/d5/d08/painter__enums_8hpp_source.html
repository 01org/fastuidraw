117  /*!
118  * enumeration to describe a gradient type.
119  */
120  enum gradient_type_t
121  {
122  /*!
123  * indicates the lack of a gradient.
124  */
125  gradient_non = 0,
126 
127  /*!
128  * Indicates a linear gradient; a linear gradient is defined
129  * by two points p0 and p1 where the interpolate at a point p
130  * is the value of dot(p - p0, p1 - p0) / dot(p0 - p1, p0 - p1).
131  */
132  gradient_linear,
133 
134  /*!
135  * Indicates a radial gradient; a radial gradient is defined
136  * by two circles C0 = Circle(p0, r0), C1 = Circle(p1, r1)
137  * where the interpolate at a point p is the time t when p
138  * is on the circle C(t) where C(t) = Circle(p(t), r(t)),
139  * p(t) = p0 + (p1 - p0) * t and r(t) = r0 + (r1 - r0) * t.
140  */
141  gradient_radial,
142 
143  /*!
144  * Indicates a sweep gradient; a sweep gradient is defined by
145  * a single point C, an angle theta (in radians), a sign S and
146  * a factor F. The angle theta represents at what angle the
147  * gradient starts, the point C is the center point of the
148  * sweep, the sign of S represents the angle orientation and
149  * the factor F reprsents how many times the gradient is to be
150  * repated. Precisely, the interpolate at a point p is defined
151  * as t_interpolate where
152  * \code
153  * vec2 d = p - C;
154  * float theta, v;
155  * theta = S * atan(d.y, d.x);
156  * if (theta < alpha )
157  * {
158  * theta += 2 * PI;
159  * }
160  * theta -= alpha;
161  * v = (theta - angle) / (2 * PI);
162  * t_interpolate = (S < 0.0) ? F * (1.0 - v) : F * v;
163  * \endcode
164  */
165  gradient_sweep,
166 
167  number_gradient_types,
168  };

341  /*!
342  * \brief
343  * Enumeration specifying blend modes. The following function-formulas
344  * are used in a number of the blend modes:
345  * \code
346  * UndoAlpha(C.rgba) = (0, 0, 0) if Ca = 0
347  * C.rgb / C.a otherwise
348  * MinColorChannel(C.rgb) = min(C.r, C.g, C.b)
349  * MaxColorChannel(C.rgb) = max(C.r, C.g, C.b)
350  * Luminosity(C.rgb) = dot(C.rgb, vec3(0.30, 0.59, 0.11))
351  * Saturation(C.rgb) = MaxColorChannel(C.rgb) - MinColorChannel(C.rgb)
352  * \endcode
353  * The next set of functions are a little messier and written in GLSL
354  * \code
355  * vec3 ClipColor(in vec3 C)
356  * {
357  * float L = Luminosity(C);
358  * float MinC = MinColorChannel(C);
359  * float MaxC = MaxColorChannel(C);
360  * if (MinC < 0.0)
361  * C = vec3(L) + (C - vec3(L)) * (L / (L - MinC));
362  * if (MaxC > 1.0)
363  * C = vec3(L) + (C - vec3(L)) * ((1 - L) / (MaxC - L));
364  * return C;
365  * }
366  *
367  * vec3 OverrideLuminosity(vec3 C, vec3 L)
368  * {
369  * float Clum = Luminosity(C);
370  * float Llum = Luminosity(L);
371  * float Delta = Llum - Clum;
372  * return ClipColor(C + vec3(Delta));
373  * }
374  *
375  * vec3 OverrideLuminosityAndSaturation(vec3 C, vec3 S, vec3 L)
376  * {
377  * float Cmin = MinColorChannel(C);
378  * float Csat = Saturation(C);
379  * float Ssat = Saturation(S);
380  * if (Csat > 0.0)
381  * {
382  * C = (C - Cmin) * Ssat / Csat;
383  * }
384  * else
