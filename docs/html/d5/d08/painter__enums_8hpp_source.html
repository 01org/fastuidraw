320  blend_w3c_multiply, /*!< W3C multiply mode: Dest * Src */
321  blend_w3c_screen, /*!< W3C screen mode: 1 - (1 - Dest) * (1 - Src) */
322 
323  /*!
324  * W3C overlay mode: for each channel,
325  * (Dst <= 0.5) ?
326  * 2.0 * Dest * Src :
327  * 1 - 2 * (1 - Dst) * (1 - Src)
328  */
329  blend_w3c_overlay,
330  blend_w3c_darken, /*!< W3C darken mode: min(Src, Dest) */
331  blend_w3c_lighten, /*!< W3C lighten mode: max(Src, Dest) */
332 
333  /*!
334  * W3C color-dodge mode: for each channel
335  * (Dest == 0) ? 0 : (Src == 1) ? 1 : min(1, Dst / (1 - Src) )
336  * i.e. if Dest is 0, write 0. If Src is 1, write 1. Otherwise
337  * write Dst / (1 - Src)
338  */
339  blend_w3c_color_dodge,
340 
341  /*!
342  * W3C color-burn mode: for each channel
343  * (Dest == 1) ? 1 : (Src == 0) ? 0 : 1 - min(1, (1 - Dst) / Src)
344  * i.e. if Dest is 1, write 1. If Src is 0, write 0. Otherwise
345  * write (1 - Dst ) / Src
346  */
347  blend_w3c_color_burn,
348 
349  /*!
350  * W3C hardlight mode: for each channel,
351  * (Src <= 0.5) ?
352  * 2.0 * Dest * Src :
353  * 1 - 2 * (1 - Dst) * (1 - Src)
354  */
355  blend_w3c_hardlight,
356 
357  /*!
358  * W3C soft light mode: for each channel:
359  * (Src <= 0.5) ?
360  * Dst - (1 - 2 * Src) * Dst * (1 - Dst) :
361  * Dst + (2 * Src - 1) * (Z - Dst)
362  * where
363  * Z = (Dst <= 0.25) ?
364  * ((16 * Dst - 12) * Dst + 4) * Dst :
365  * sqrt(Dst)
366  */
367  blend_w3c_softlight,
368 
369  blend_w3c_difference, /*!< W3C difference mode: for each channel, abs(Dest - Src) */
370  blend_w3c_exclusion, /*!< W3C exclusion mode: for each channel, Dest + Src - 2 * Dest * Src */
