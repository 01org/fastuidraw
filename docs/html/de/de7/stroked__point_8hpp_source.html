32 /*!
33  * \brief
34  * A StrokedPoint holds the data for a point of stroking
35  * where all segments are line segments (coming from
36  * curve tessellation). The upshot is that the fragment
37  * shader does NOT perform any coverage computation for
38  * non-dashed stroking. In addition, the data is so that
39  * changing the stroking width or miter limit does not
40  * change the stroking data.
41  */
42 class StrokedPoint
43 {
44 public:
45  /*!
46  * \brief
47  * Enumeration for specifing the point type which in turn
48  * determines the meaning of the fields \ref m_pre_offset
49  * and \ref m_auxiliary_offset
50  */
51  enum offset_type_t
52  {
53  /*!
54  * The point is for an edge of the path, point signifies the start
55  * of a sub-edge (quad) of drawing an edge. The meanings of \ref
56  * m_pre_offset and \ref m_auxiliary_offset are:
57  * - \ref m_pre_offset the normal vector to the edge in which
58  * move the point by when stroking
59  * - \ref m_auxiliary_offset when added to \ref m_position, gives
60  * the position of the point on the other
61  * side of the edge.
62  */
63  offset_sub_edge,
64 
65  /*!
66  * The point is at a position that has the same value as point on
67  * an edge but the point itself is part of a cap or join. The
68  * meanings of the members \ref m_pre_offset and \ref
69  * m_auxiliary_offset are:
70  * - \ref m_pre_offset the normal vector to the edge in which move
71  * the point by when stroking; this vector can
72  * be (0, 0).
73  * - \ref m_auxiliary_offset unused (set to (0, 0))
74  */
75  offset_shared_with_edge,
76 
77  /*!
78  * The point is for a boundary point of a rounded join of the path.
79  * The meanings of the members \ref m_pre_offset and and \ref
80  * m_auxiliary_offset are:
81  * - \ref m_pre_offset the .x() component holds the unit normal vector
82  * between the join point and the edge going into
83  * the join. The .y() component holds the unit
84  * normal vector between the join point and the edge
85  * leaving the join. The packing is that the
86  * x-coordinate value is given and the y-coordinate
87  * magnitude is \f$ sqrt(1 - x^2) \f$. If the bit
88  * \ref normal0_y_sign_bit is up, then the y-coordinate
89  * for the normal vector of going into the join,
90  * then the y-value is negative. If the bit \ref
91  * normal1_y_sign_bit is up, then the y-coordinate
92  * for the normal vector of leaving the join, then
93  * the y-value is negative.
94  * - \ref m_auxiliary_offset the .x() component gives an interpolation
95  * in the range [0, 1] to interpolate between
96  * the normal vectors packed in \ref m_pre_offset.
97  * The .y() value gives the normal vector directly
98  * but packed (as in \ref m_pre_offset) where the
99  * y-coordinate sign is negative if the bit \ref
100  * sin_sign_mask is up.
101  */
102  offset_rounded_join,
103 
104  /*!
105  * Point type for miter-clip join point whose position depends on the
106  * stroking radius and the miter-limit. The meanings of \ref m_pre_offset
107  * and \ref m_auxiliary_offset are:
108  * - \ref m_pre_offset gives the unit normal vector of teh edge going
109  * into the join
110  * - \ref m_auxiliary_offset gives the unit normal vector of teh edge
111  * leaving the join
112  */
113  offset_miter_clip_join,
114 
115  /*!
116  * Point type for miter-bevel join point whose position depends on the
117  * stroking radius and the miter-limit. The meanings of \ref m_pre_offset
118  * and \ref m_auxiliary_offset are:
119  * - \ref m_pre_offset gives the unit normal vector of teh edge going
120  * into the join
121  * - \ref m_auxiliary_offset gives the unit normal vector of teh edge
122  * leaving the join
123  */
124  offset_miter_bevel_join,
125 
126  /*!
127  * Point type for miter join whose position position depends on the
128  * stroking radius and the miter-limit. The meanings of \ref m_pre_offset
129  * and \ref m_auxiliary_offset are:
130  * - \ref m_pre_offset gives the unit normal vector of teh edge going
131  * into the join
132  * - \ref m_auxiliary_offset gives the unit normal vector of teh edge
133  * leaving the join
134  */
135  offset_miter_join,
136 
137  /*!
138  * The point is for a boundary point of a rounded cap of the path.
139  * The meanings of \ref m_pre_offset and \ref m_auxiliary_offset
140  * are:
141  * - \ref m_pre_offset is the normal vector to the path to start
142  * drawing the rounded cap
143  * - \ref m_auxiliary_offset gives the the unit vector (cos, sin)
144  * of the angle to make with the vector
145  * given by \ref m_pre_offset
146  */
147  offset_rounded_cap,
148 
149  /*!
150  * The point is for a boundary point of a square cap of the path,
151  * the meanings of \ref m_pre_offset and \ref m_auxiliary_offset are:
152  * - \ref m_pre_offset is the normal vector to the path by which
153  * to move the point
154  * - \ref m_auxiliary_offset is the tangent vector to the path
155  * by which to move the point
156  */
157  offset_square_cap,
158 
159  /*!
160  * The point is a point of an adjustable cap. These points are for
161  * dashed stroking with caps; they contain data to allow one from a
162  * vertex shader to extend or shrink the cap area correctly to implement
163  * dashed stroking. The meanings of \ref m_pre_offset and \ref
164  * m_auxiliary_offset are:
165  * - \ref m_pre_offset is the normal vector to the path by which
166  * to move the point; this value can be (0, 0)
167  * to indicate to not move perpindicular to the
168  * path
169  * - \ref m_auxiliary_offset is the tangent vector to the path
170  * by which to move the point; this value
171  * can be (0, 0) to indicate to not move
172  * parallel to the path
173  */
174  offset_adjustable_cap,

179  number_offset_types
