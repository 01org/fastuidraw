35 /*!
36  * \brief
37  * A StrokedPoint holds the data for a point of stroking
38  * where all segments are line segments (coming from
39  * curve tessellation). The upshot is that the fragment
40  * shader does NOT perform any coverage computation for
41  * non-dashed stroking. In addition, the data is so that
42  * changing the stroking width or miter limit does not
43  * change the stroking data.
44  */

54  enum offset_type_t
55  {
56  /*!
57  * The point is for an edge of the path, point signifies the start
58  * or end of a sub-edge (quad) of drawing an edge. The meanings of

60  * - \ref m_pre_offset the normal vector to the edge in which

64  * side of the edge.
65  */

67 
68  /*!
69  * The point is at a position that has the same value as point on
70  * an edge but the point itself is part of a cap or join. The
71  * meanings of the members \ref m_pre_offset and \ref
72  * m_auxiliary_offset are:
73  * - \ref m_pre_offset the normal vector to the edge in which move
74  * the point by the stroking width when stroking;
75  * this vector can be (0, 0).
76  * - \ref m_auxiliary_offset unused (set to (0, 0))
77  */
78  offset_shared_with_edge,
79 
80  /*!
81  * The point is for a boundary point of a rounded join of the path.
82  * The meanings of the members \ref m_pre_offset and and \ref
83  * m_auxiliary_offset are:
84  * - \ref m_pre_offset the .x() component holds the unit normal vector
85  * between the join point and the edge going into
86  * the join. The .y() component holds the unit
87  * normal vector between the join point and the edge
88  * leaving the join. The packing is that the
89  * x-coordinate value is given and the y-coordinate
90  * magnitude is \f$ sqrt(1 - x^2) \f$. If the bit
91  * \ref normal0_y_sign_bit is up, then the y-coordinate
92  * for the normal vector of going into the join,
93  * then the y-value is negative. If the bit \ref
94  * normal1_y_sign_bit is up, then the y-coordinate
95  * for the normal vector of leaving the join, then
96  * the y-value is negative.
97  * - \ref m_auxiliary_offset the .x() component gives an interpolation
98  * in the range [0, 1] to interpolate between
99  * the normal vectors packed in \ref m_pre_offset.
100  * The .y() value gives the normal vector directly
101  * but packed (as in \ref m_pre_offset) where the
102  * y-coordinate sign is negative if the bit \ref
103  * sin_sign_mask is up.
104  */
105  offset_rounded_join,
106 
107  /*!
108  * Point type for miter-clip join point whose position depends on the
109  * stroking radius and the miter-limit. The meanings of \ref m_pre_offset
110  * and \ref m_auxiliary_offset are:
111  * - \ref m_pre_offset gives the unit normal vector of the edge going

113  * - \ref m_auxiliary_offset gives the unit normal vector of the edge
114  * leaving the join
115  */
