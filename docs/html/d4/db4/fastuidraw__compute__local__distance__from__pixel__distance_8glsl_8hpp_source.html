24 /*!
25  * Given a point (clip_p) and a direction (clip_direction) in
26  * clip-coordinates, compute the coeffecient W so that AFTER
27  * projection, the points clip_p and (clip_p + W * clip_direction)
28  * are pixel_distance pixels apart.
29  * \param pixel_distance the distance in pixels to move by
30  * \param clip_p the position of the vertex in CLIP coordiantes
31  * \param clip_direction the direction by which to move in CLIP coordiantes
32  */
33 float
34 fastuidraw_local_distance_from_pixel_distance(in float pixel_distance,
35  in vec3 clip_p,
36  in vec3 clip_direction)
37 {
38  vec2 zeta;
39  vec3 p, v;
40  float return_value, length_zeta;
41 
42  /* convert clip-coordinates so that after dividing by .z,
43  * the coordinates are in -pixel- units.
44  */
45  p = vec3(0.5 * fastuidraw_viewport_pixels.xy, 1.0) * clip_p;
46  v = vec3(0.5 * fastuidraw_viewport_pixels.xy, 1.0) * clip_direction;
47 
48  /* From where this formula comes. Our challenge is to give a stroking width
49  * in local (pre-projection) coordiantes from a stroking width of pixels.
50  * We have a point p (given by position) and a direction v (given by offset)
51  * and we wish to find an L so that after projection p + L*v is W pixels
52  * from p. Now for some math.
53  *
54  * Let m>0, and let q(m) = p + m*v. Once m>0 is large enough, we can
55  * project q(m) and p to the screen, find the screen space barycentric
56  * W pixels from p.
57  *
58  * Let P = K * projection(p)
59  * Let Q(m) = K * projection(q(m))
60  *
61  * where
62  *
63  * K = (0.5 * fastuidraw_viewport_pixels.x, 0.5 * fastuidraw_viewport_pixels.y, 1)
64  * projection(x,y,z) = (x/z, y/z)
65  *
66  * Let R = P + W * DELTA where DELTA = (Q(m) - P)/||Q(m)-P||
67  * Let s = W / ||Q(m) - P|| gives that P + s * (Q(m) - P) = R.
68  * Let t = ( s/q(m)_z ) / ( (1-s)/p_z + s/q(m)_z). Then
69  * projection( p + t * (q(m) - p) ) = projection( p + (t*m) * v) = R
70  *
71  * Thus the coefficient we want is given by (t*m).
72  *
73  * t * m = ( m * s * P_z) / ( (1-s)Q(m)_z + s * P_z)
74  *
75  * which simplifies to (after lots of algebra) to
76  *
77  * t * m = (W * p_z * p_z) / (- W * p_z * |v_z| + ||zeta||)
78  *
79  * where zeta = (v_x * p_z - v_z * p_x, v_y * p_z - v_z * p_y)
80  */
81  zeta = vec2(v.x * p.z - v.z * p.x,
82  v.y * p.z - v.z * p.y);
83  length_zeta = length(zeta);
84 
85  return_value = pixel_distance * p.z * p.z;
86  return_value /= (-pixel_distance * abs(p.z * v.z) + length_zeta);
87 
88  return return_value;
89 }
