35  * A shareable_value_list is a list of values and their types

38  * from a dependee shader.

40  class shareable_value_list

43  /*!
44  * \brief
45  * Enumeration to define the types of the a shareable value

47  */
48  enum type_t
49  {
50  type_float, /*!< corresponds to float in GLSL */
51  type_uint, /*!< corresponds to uint in GLSL */
52  type_int, /*!< corresponds to int in GLSL */
53 
54  type_number_types,
55  };
56 
57  /*!
58  * Swap operation
59  * \param obj object with which to swap
60  */
61  void
62  swap(shareable_value_list &obj)
63  {
64  m_data.swap(obj.m_data);
65  }
66 
67  /*!
68  * Returns an array indexed by \ref type_t that
69  * holds the number of shareable variables for
70  * each type.
71  */
72  vecN<unsigned int, type_number_types>
73  number_shareable_values(void) const
74  {
75  vecN<unsigned int, type_number_types> R;
76  for (unsigned int i = 0; i < type_number_types; ++i)
77  {
78  R[i] = m_data[i].size();
79  }
80  return R;
81  }
