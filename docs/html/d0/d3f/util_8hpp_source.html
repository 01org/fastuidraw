307  /*!
308  * Pack the lowest N bits of a value at a bit.
309  * \param bit0 bit location of return value at which to pack
310  * \param num_bits number of bits from value to pack
311  * \param value value to pack
312  */
313  inline
314  uint32_t
315  pack_bits(uint32_t bit0, uint32_t num_bits, uint32_t value)
316  {
317  uint32_t mask;
318  mask = (1u << num_bits) - 1u;
319  FASTUIDRAWassert(bit0 + num_bits <= 32u);
320  FASTUIDRAWassert(value <= mask);
321  return (value & mask) << bit0;
322  }
