174  /*!
175  * \brief
176  * A PainterItemShaderGLSL is a collection of GLSL source code
177  * fragments for a \ref PainterShaderRegistrarGLSL.
178  *
179  * The vertex shader code needs to implement the function:
180  * \code
181  * void
182  * fastuidraw_gl_vert_main(in uint sub_shader,
183  * in uvec4 attrib0,
184  * in uvec4 attrib1,
185  * in uvec4 attrib2,
186  * in uint shader_data_offset,
187  * out uint z_add,
188  * out vec2 brush_p,
189  * out vec3 clip_p)
190  * \endcode
191  * where
192  * - sub_shader corresponds to PainterItemShader::sub_shader()
193  * - attrib0 corresponds to PainterAttribute::m_attrib0,
194  * - attrib1 corresponds to PainterAttribute::m_attrib1,
195  * - attrib2 corresponds to PainterAttribute::m_attrib2 and
196  * - shader_data_offset is what block in the data store for
197  * the data packed by PainterItemShaderData::pack_data()
198  * of the PainterItemShaderData in the \ref Painter call;
199  * use the macro fastuidraw_fetch_data() to read the data.
200  *
201  * The output clip_p is to hold the clip-coordinate of the vertex.
202  * The output brush_p is to hold the coordinate for the brush of
203  * the vertex. The out z_add must be written to as well and it
204  * is how much to add to the value in \ref PainterHeader::m_z
205  * for the purpose of intra-item z-occluding. Items that do
206  * not self-occlude should write 0 to z_add.
207  *
208  * The fragment shader code needs to implement the function:
209  * \code
210  * vec4
211  * fastuidraw_gl_frag_main(in uint sub_shader,
212  * in uint shader_data_offset)
213  * \endcode
214  *
215  * which returns the color of the fragment for the item -before-
216  * the color modulation by the pen, brush or having compositing
217  * applied. In addition, the color value returned is NOT
218  * pre-multiplied by alpha either.
219  *
220  * Available to only the vertex shader are the following:
221  * - mat3 fastuidraw_item_matrix (the 3x3 matrix from item coordinate to clip coordinates)
222  *
223  * Available to both the vertex and fragment shader are the following:
224  * - sampler2DArray fastuidraw_imageAtlasLinear the color texels (AtlasColorBackingStoreBase) for images unfiltered
225  * - sampler2DArray fastuidraw_imageAtlasLinearFiltered the color texels (AtlasColorBackingStoreBase) for images bilinearly filtered
226  * - usampler2DArray fastuidraw_imageIndexAtlas the texels of the index atlas (AtlasIndexBackingStoreBase) for images
227  * - the macro fastuidraw_fetch_data(B) to fetch the B'th block from the data store buffer
228  * (PainterDraw::m_store), return as uvec4. To get floating point data use, the GLSL
229  * built-in function uintBitsToFloat().
230  * - the macro fastuidraw_fetch_glyph_data(B) to read the B'th value from the glyph data
231  * (GlyphAtlasBackingStoreBase), return as uint. To get floating point data, use the GLSL
232  * built-in function uintBitsToFloat().
233  * - the macro fastuidraw_colorStopFetch(x, L) to retrieve the color stop value at location x of layer L
234  * - vec2 fastuidraw_viewport_pixels the viewport dimensions in pixels
235  * - vec2 fastuidraw_viewport_recip_pixels reciprocal of fastuidraw_viewport_pixels
236  * - vec2 fastuidraw_viewport_recip_pixels_magnitude euclidean length of fastuidraw_viewport_recip_pixels
237  *
238  * For both stages, the value of the argument of shader_data_offset is which block into the data
239  * store (PainterDraw::m_store) of the custom shader data. Do
240  * \code
241  * fastuidraw_fetch_data(shader_data_offset)
242  * \endcode
243  * to read the raw bits of the data. The type returned by the macro fastuidraw_fetch_data()
244  *
245  * Use the GLSL built-in uintBitsToFloat() to covert the uint bit-value to float
246  * and just cast int() to get the value as an integer.
247  *
248  * Also, if one defines macros in any of the passed ShaderSource objects,
249  * those macros MUST be undefined at the end. In addition, if one
250  * has local helper functions, to avoid global name collision, those
251  * function names should be wrapped in the macro FASTUIDRAW_LOCAL()
252  * to make sure that the function is given a unique global name within
253  * the uber-shader.
254  *
255  * Lastly, one can use the class \ref UnpackSourceGenerator to generate
256  * shader code to unpack values from the data in the data store buffer.
257  * That machine generated code uses the macro fastuidraw_fetch_data().
258  */
259  class PainterItemShaderGLSL:public PainterItemShader
260  {
261  public:
262  /*!
263  * Ctor.
264  * \param puses_discard set to true if and only if the shader code

268  * \param fragment_src GLSL source holding fragment shader routine
269  * \param varyings list of varyings of the shader
270  * \param num_sub_shaders the number of sub-shaders it supports

272  PainterItemShaderGLSL(bool puses_discard,
