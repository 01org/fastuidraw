89  /*!
90  * Compute pixel coordinates from normalized device coords
91  * using this Viewport values. The pixel coordinates are so
92  * that (0, 0) is the bottom left.
93  * \param ndc normalized device coordinates
94  */
95  vec2
96  compute_pixel_coordinates(vec2 ndc) const
97  {
98  ndc += vec2(1.0f); // place in range [0, 2]
99  ndc *= 0.5f; // normalize to [0, 1]

145  /*!
146  * Compute normalized device coordinates from pixel
147  * coordinates.
148  * \param pixel pixel coordinates where (0, 0) corresponds to bottom
149  * left of the surface
150  */
151  vec2
152  compute_normalized_device_coords(vec2 pixel) const
153  {
154  pixel -= vec2(m_origin); // translate from origin
155  pixel /= vec2(m_dimensions); // normalize to [0, 1]
156  pixel *= 2.0f; // normalize to [0, 2]
157  pixel -= vec2(1.0f); // palce in range [-1, 1]
158  return pixel;
