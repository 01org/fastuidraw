208  /*!
209  * Ctor. Initializes as valid with blending on,
210  * with blend equation as add in all channels,
211  * with src func in all channels as \ref ONE
212  * and dest func in all channels as \ref ZERO.
213  */
214  BlendMode(void)
215  {
216  m_value = pack_bits(blending_on_bit, 1u, 1u)
217  | pack_bits(equation_rgb_bit0, equation_num_bits, ADD)
218  | pack_bits(equation_alpha_bit0, equation_num_bits, ADD)
219  | pack_bits(src_func_rgb_bit0, func_num_bits, ONE)
220  | pack_bits(src_func_alpha_bit0, func_num_bits, ONE)
221  | pack_bits(dst_func_rgb_bit0, func_num_bits, ZERO)
222  | pack_bits(dst_func_alpha_bit0, func_num_bits, ZERO);
223  }
